166  /*
167  * Each bucket stores two elements:
168  * - An aligned storage to store a value_type object with placement-new
169  * - An unsigned integer of type neighborhood_bitmap used to tell us which buckets in the neighborhood of the
170  * current bucket contain a value with a hash belonging to the current bucket.
171  *
172  * For a bucket 'b' a bit 'i' (counting from 0 and from the least significant bit to the most significant)
173  * set to 1 means that the bucket 'b+i' contains a value with a hash belonging to bucket 'b'.
174  * The bits used for that, start from the third least significant bit.
175  *
176  * The least significant bit is set to 1 if there is a value in the bucket storage.
177  * The second least significant bit is set to 1 if there is an overflow. More than NeighborhoodSize values
178  * give the same hash, all overflow values are stored in the m_overflow_elements list of the map.

887  template<typename U = value_type,
888  typename std::enable_if<std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
889  void rehash_internal(size_type count) {
890  /*
891  * Little more complicate here. We want to avoid any exception so we can't just insert elements
892  * in the map trough insert_internal. The method may throw even if value_type is nothrow_move_constructible
893  * due to the m_overflow_elements.push_back in the method.
894  *
895  * So we first move safely all the elements from m_buckets in the new map (we know they will not go into
896  * overflow). We then swap m_overflow_elements into the new map and we update all necessary overflow flags.
897  *
898  * This way we don't do any memory allocation (outside the construction of m_buckets) and we avoid
899  * any exception which may hinder the strong exception-safe guarantee.
900  */
901  hopscotch_hash tmp_map(count, m_hash, m_key_equal, get_allocator(), m_max_load_factor);
902 
903  for(hopscotch_bucket& bucket : m_buckets) {
904  if(bucket.is_empty()) {
905  continue;
906  }
907 
908  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(
909  tmp_map.m_hash(KeySelect()(bucket.get_value())));
910  tmp_map.insert_internal(std::move(bucket.get_value()), ibucket_for_hash);
911  }
912 
913  tsl_assert(tmp_map.m_overflow_elements.empty());
914  if(!m_overflow_elements.empty()) {
915  tmp_map.m_overflow_elements.swap(m_overflow_elements);
916  tmp_map.m_nb_elements += tmp_map.m_overflow_elements.size();
917 
918  for(const value_type& value : tmp_map.m_overflow_elements) {
919  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(KeySelect()(value)));
920  tmp_map.m_buckets[ibucket_for_hash].set_overflow(true);
921  }
922  }
923 
924  tmp_map.swap(*this);
925  }

927  /*
928  * Find in m_overflow_elements an element for which the bucket it initially belongs to,
929  * equals original_bucket_for_hash.
930  *
931  * Return m_overflow_elements.end() if none.
932  */
933  iterator_overflow find_in_overflow_from_bucket(iterator_overflow search_start,
934  std::size_t original_bucket_for_hash)
935  {
936  for(auto it = search_start; it != m_overflow_elements.end(); ++it) {
937  const std::size_t bucket_for_overflow_hash = bucket_for_hash(m_hash(KeySelect()(*it)));
938  if(bucket_for_overflow_hash == original_bucket_for_hash) {
939  return it;
940  }
941  }
942 
943  return m_overflow_elements.end();
944  }

1018  std::pair<iterator, bool> insert_internal(P&& value, std::size_t ibucket_for_hash) {
1019  if((m_nb_elements - m_overflow_elements.size() + 1) > m_load_threshold) {
1020  rehash_internal(get_expand_size());
1021  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
1022  }
1023 
1024  std::size_t ibucket_empty = find_empty_bucket(ibucket_for_hash);
1025  if(ibucket_empty < m_buckets.size()) {
1026  do {
1027  // Empty bucket is in range of NeighborhoodSize, use it
1028  if(ibucket_empty - ibucket_for_hash < NeighborhoodSize) {
1029  auto it = insert_in_bucket(std::forward<P>(value), ibucket_empty, ibucket_for_hash);
1030  return std::make_pair(iterator(it, m_buckets.end(), m_overflow_elements.begin()), true);
1031  }
1032  }
1033  // else, try to swap values to get a closer empty bucket
1034  while(swap_empty_bucket_closer(ibucket_empty));
1035  }

1038  if(!will_neighborhood_change_on_rehash(ibucket_for_hash)) {
1039  m_overflow_elements.push_back(std::forward<P>(value));
1040  m_buckets[ibucket_for_hash].set_overflow(true);
1041  m_nb_elements++;
1042

1044  std::prev(m_overflow_elements.end())), true);
1045  }
1046 
1047  rehash_internal(get_expand_size());
1048 
1049  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
1050  return insert_internal(std::forward<P>(value), ibucket_for_hash);
1051  }
