738  template<typename U = value_type, typename std::enable_if<std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
739  void rehash_internal(size_type count) {
740  /*
741  * Little more complicate here. We want to avoid any exception so we can't just insert elements
742  * in the map trough insert_internal. The method may throw even if value_type is nothrow_move_constructible
743  * due to the m_overflow_elements.push_back in the method.
744  *
745  * So we first move safely all the elements from m_buckets in the new map (we know they will not go into overflow).
746  * We then swap m_overflow_elements into the new map and we update all necessary overflow flags.
747  *
748  * This way we don't do any memory allocation (outside the construction of m_buckets) and we avoid
749  * any exception which may hinder the strong exception-safe guarantee.
750  */
751  hopscotch_hash tmp_map(count, m_hash, m_key_equal, get_allocator(), m_max_load_factor);
752 
753  for(hopscotch_bucket& bucket : m_buckets) {
754  if(bucket.is_empty()) {
755  continue;
756  }
757 
758  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(KeySelect()(bucket.get_value_type())));
759  tmp_map.insert_internal(std::move(bucket.get_value_type()), ibucket_for_hash);
760  }
761 
762  assert(tmp_map.m_overflow_elements.empty());
763  if(!m_overflow_elements.empty()) {
764  tmp_map.m_overflow_elements.swap(m_overflow_elements);
765  tmp_map.m_nb_elements += tmp_map.m_overflow_elements.size();
766 
767  for(const value_type& value : tmp_map.m_overflow_elements) {
768  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(KeySelect()(value)));
769  tmp_map.m_buckets[ibucket_for_hash].set_overflow(true);
770  }
771  }
772 
773  std::swap(*this, tmp_map);
774  }

848  template<typename P>
849  std::pair<iterator, bool> insert_internal(P&& value, std::size_t ibucket_for_hash) {
850  assert(!m_buckets.empty());
851 
852  if((m_nb_elements - m_overflow_elements.size() + 1) > m_load_threshold) {
853  rehash_internal(get_expand_size());
854  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
855  }
856 
857  std::size_t ibucket_empty = find_empty_bucket(ibucket_for_hash);
858  if(ibucket_empty < m_buckets.size()) {
859  do {
860  // Empty bucket is in range of NeighborhoodSize, use it
861  if(ibucket_empty - ibucket_for_hash < NeighborhoodSize) {
862  auto it = insert_in_bucket(std::forward<P>(value), ibucket_empty, ibucket_for_hash);
863  return std::make_pair(iterator(it, m_buckets.end(), m_overflow_elements.begin()), true);
864  }
865  }
866  // else, try to swap values to get a closer empty bucket
867  while(swap_empty_bucket_closer(ibucket_empty));
868  }
869 
870  // A rehash will not change the neighborhood, put the value in overflow list
871  if(!will_neighborhood_change_on_rehash(ibucket_for_hash)) {
872  m_overflow_elements.push_back(std::forward<P>(value));
873  m_buckets[ibucket_for_hash].set_overflow(true);
874  m_nb_elements++;
875 
876  return std::make_pair(iterator(m_buckets.end(), m_buckets.end(), std::prev(m_overflow_elements.end())), true);
877  }
878 
879  rehash_internal(get_expand_size());
880 
881  ibucket_for_hash = bucket_for_hash(m_hash(KeySelect()(value)));
882  return insert_internal(std::forward<P>(value), ibucket_for_hash);
883  }

940  *
941  * If a swap was possible, the position of ibucket_empty_in_out will be closer to 0 and true will re returned.
942  */
943  bool swap_empty_bucket_closer(std::size_t& ibucket_empty_in_out) {
944  assert(ibucket_empty_in_out >= NeighborhoodSize);
945  const std::size_t neighborhood_start = ibucket_empty_in_out - NeighborhoodSize + 1;
946

949  std::size_t to_swap = to_check;
950 
951  while(neighborhood_infos != 0 && to_swap < ibucket_empty_in_out) {
952  if((neighborhood_infos & 1) == 1) {
953  assert(m_buckets[ibucket_empty_in_out].is_empty());
954  assert(!m_buckets[to_swap].is_empty());
955

957

959  assert(m_buckets[to_check].check_neighbor_presence(to_swap - to_check));
960 
961  m_buckets[to_check].toggle_neighbor_presence(ibucket_empty_in_out - to_check);
962  m_buckets[to_check].toggle_neighbor_presence(to_swap - to_check);

966

968  }

976  }
