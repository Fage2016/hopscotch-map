138  /*
139  * Each bucket stores two elements:
140  * - An aligned storage to store a value_type object with placement-new
141  * - An unsigned integer of type neighborhood_bitmap used to tell us which buckets in the neighborhood of the current bucket
142  * contain a value with a hash belonging to the current bucket.
143  *
144  * For a bucket 'b' a bit 'i' (counting from 0 and from the least significant bit to the most significant)
145  * set to 1 means that the bucket 'b+i' contains a value with a hash belonging to bucket 'b'.
146  * The bits used for that, start from the third least significant bit.
147  *
148  * The least significant bit is set to 1 if there is a value in the bucket storage.
149  * The second least significant bit is set to 1 if there is an overflow. More than NeighborhoodSize values give the same hash,
150  * all overflow values are stored in the m_overflow_elements list of the map.
151  */

796  template<typename U = value_type, typename std::enable_if<std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
797  void rehash_internal(size_type count) {
798  /*
799  * Little more complicate here. We want to avoid any exception so we can't just insert elements
800  * in the map trough insert_internal. The method may throw even if value_type is nothrow_move_constructible
801  * due to the m_overflow_elements.push_back in the method.
802  *
803  * So we first move safely all the elements from m_buckets in the new map (we know they will not go into overflow).
804  * We then swap m_overflow_elements into the new map and we update all necessary overflow flags.
805  *
806  * This way we don't do any memory allocation (outside the construction of m_buckets) and we avoid
807  * any exception which may hinder the strong exception-safe guarantee.
808  */
809  hopscotch_map tmp_map(count, m_hash, m_key_equal, get_allocator(), m_max_load_factor);
810 
811  for(hopscotch_bucket & bucket : m_buckets) {
812  if(bucket.is_empty()) {
813  continue;
814  }
815 
816  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(bucket.get_key_value().first));
817  tmp_map.insert_internal(std::move(bucket.get_key_value()), ibucket_for_hash);
818  }
819 
820  assert(tmp_map.m_overflow_elements.empty());
821  if(!m_overflow_elements.empty()) {
822  tmp_map.m_overflow_elements.swap(m_overflow_elements);
823  tmp_map.m_nb_elements += tmp_map.m_overflow_elements.size();
824 
825  for(const value_type & key_value : tmp_map.m_overflow_elements) {
826  const std::size_t ibucket_for_hash = tmp_map.bucket_for_hash(tmp_map.m_hash(key_value.first));
827  tmp_map.m_buckets[ibucket_for_hash].set_overflow(true);
828  }
829  }
830 
831  std::swap(*this, tmp_map);
832  }

836  * Return m_overflow_elements.end() if none.
837  */
838  typename std::list<value_type>::iterator find_in_overflow_from_bucket(typename std::list<value_type>::iterator search_start,
839  std::size_t original_bucket_for_hash)

845  }

1023  assert(!m_buckets[to_swap].is_empty());
