104  /*
105  * Each bucket store two elements:
106  * - An aligned storage to store a value_type object with placement-new
107  * - An unsigned integer of type neighborhood_bitmap used to tell us which buckets in the neighborhood of the current bucket
108  * contain a value with a hash belonging to the current bucket.
109  *
110  * For a bucket 'b' a bit 'i' (counting from 0 and from the least significant bit to the most significant)
111  * set to 1 means that the bucket 'b+i' contains a value with a hash belonging to bucket 'b'.
112  * The bits used for that, start from the third least significant bit.
113  *
114  * The least significant bit is set to 1 if there is a value in the bucket storage.
115  * The second least significant bit is set to 1 if there is an overflow. More than NeighborhoodSize values give the same hash,
116  * all overflow values are stored in the m_overflow_elements list of the map.

669  return std::make_pair(it_find, false);
670  }
671 
672 
673  std::size_t ibucket_empty = find_empty_bucket(ibucket_for_hash);
674  if(ibucket_empty < m_buckets.size()) {
675  do {
676  // Empty bucket is in range of NeighborhoodSize, use it
677  if(ibucket_empty - ibucket_for_hash < NeighborhoodSize) {
678  auto it = insert_in_bucket(std::forward<P>(key_value), ibucket_empty, ibucket_for_hash);
679  return std::make_pair(iterator(it, m_buckets.end(), m_overflow_elements.begin()), true);
680  }
681  }
682  // else, try to swap values to get a closer empty bucket
683  while(find_swap(ibucket_empty));
684 
685 
686  // A rehash will not change the neighborhood, put the value in overflow list
687  if(!will_neighborhood_change_on_rehash(ibucket_for_hash)) {
688  m_overflow_elements.push_back(std::forward<P>(key_value));
689  m_buckets[ibucket_for_hash].set_overflow(true);
690  m_nb_elements++;
691 
692  return std::make_pair(iterator(m_buckets.end(), m_buckets.end(), std::prev(m_overflow_elements.end())), true);
693  }
694 
695 
696  }
697 
698  rehash(m_buckets.size() * REHASH_SIZE_MULTIPLICATION_FACTOR);
699  return insert_internal(std::forward<P>(key_value));
700  }
701 
702  /*
703  * Return true if a rehash will change the position of a key-value in the neighborhood of ibucket_neighborhood_check.
704  * In this case a rehash is needed instead of puting the value in overflow list.
705  */
706  bool will_neighborhood_change_on_rehash(size_t ibucket_neighborhood_check) {
707  for(size_t ibucket = ibucket_neighborhood_check;
708  ibucket < m_buckets.size() && (ibucket - ibucket_neighborhood_check) < NeighborhoodSize;
709  ++ibucket)
710  {
711  assert(!m_buckets[ibucket].is_empty());
712 
713  const value_type & key_value = m_buckets[ibucket].get_key_value();
714  const size_t hash = m_hash(key_value.first);
715 
716  if((hash & (m_buckets.size() - 1)) != (hash & (m_buckets.size() * REHASH_SIZE_MULTIPLICATION_FACTOR - 1))) {
717  return true;
718  }
719  }
720 
721  return false;
722  }

726  * If none, the returned index equals m_buckets.size()

729  for(; ibucket_start < m_buckets.size(); ibucket_start++) {

731  break;

738  /*
739  * Insert key_value in ibucket_empty for key_value originally belonging to ibucket_for_hash
740  *
741  * Return bucket iterator to ibucket_empty
742  */
743  template<typename P>
744  typename std::vector<hopscotch_bucket>::iterator insert_in_bucket(P && key_value, std::size_t ibucket_empty, std::size_t ibucket_for_hash) {
745  assert(ibucket_empty >= ibucket_for_hash );
746  assert(m_buckets[ibucket_empty].is_empty());
747  m_buckets[ibucket_empty].set_key_value(std::forward<P>(key_value));
748 
749  assert(!m_buckets[ibucket_for_hash].is_empty());
750  m_buckets[ibucket_for_hash].toggle_neighbor_presence(ibucket_empty - ibucket_for_hash);
751  m_nb_elements++;
752 
753  return m_buckets.begin() + ibucket_empty;
754  }
755 
756  /*
757  * Try to swap the bucket ibucket_empty_in_out with a bucket preceding it while keeping the neighborhood conditions correct.
758  *
759  * If a swap was possible, the position of ibucket_empty_in_out will be closer to 0 and true will re returned.
760  */
761  bool find_swap(std::size_t & ibucket_empty_in_out) {
762  assert(ibucket_empty_in_out >= NeighborhoodSize);
763  const std::size_t neighborhood_start = ibucket_empty_in_out - NeighborhoodSize + 1;
764 
765  for(std::size_t to_check = neighborhood_start; to_check < ibucket_empty_in_out; to_check++) {
766  neighborhood_bitmap neighborhood_infos = m_buckets[to_check].get_neighborhood_infos();
767  std::size_t to_swap = to_check;
768 
769  while(neighborhood_infos != 0 && to_swap < ibucket_empty_in_out) {
770  if((neighborhood_infos & 1) == 1) {
771  assert(m_buckets[ibucket_empty_in_out].is_empty());
772  assert(!m_buckets[to_swap].is_empty());
773 
774  m_buckets[to_swap].swap_key_value_with_empty_bucket(m_buckets[ibucket_empty_in_out]);
775 
776  assert(!m_buckets[to_check].check_neighbor_presence(ibucket_empty_in_out - to_check));
777  assert(m_buckets[to_check].check_neighbor_presence(to_swap - to_check));
778 
779  m_buckets[to_check].toggle_neighbor_presence(ibucket_empty_in_out - to_check);
780  m_buckets[to_check].toggle_neighbor_presence(to_swap - to_check);
781 
782 
783  ibucket_empty_in_out = to_swap;
784 
785  return true;
786  }
787 
788  to_swap++;
789  neighborhood_infos = (neighborhood_bitmap) (neighborhood_infos >> 1);
790  }
791  }
792 
793  return false;
794  }
